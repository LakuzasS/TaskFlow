def send_email(subject, body, to):
    """
        Envoie un email à un destinataire avec un sujet et un corps donnés.

        Cette fonction utilise un compte de messagerie parmi ceux définis dans la liste `accounts`.
        Elle sélectionne le compte suivant dans la liste à chaque envoi, en utilisant un mécanisme de rotation.

        :param subject: Sujet de l'email.
        :param body: Corps de l'email.
        :param to: Adresse email du destinataire.
    """
    global current_account_index
    account = accounts[current_account_index]

    try:
        with smtplib.SMTP("smtp.gmail.com", 587) as server:
            server.starttls()
            server.login(account["email"], account["password"])

            msg = MIMEMultipart()
            msg["From"] = account["email"]
            msg["To"] = to
            msg["Subject"] = subject

            msg.attach(MIMEText(body, "plain", "utf-8"))

            server.sendmail(account["email"], to, msg.as_string())
            print(f'Email envoyé à {to} avec succès depuis {account["email"]}.')
            current_account_index = (current_account_index + 1) % len(accounts)

    except Exception as e:
        print(f"Erreur lors de l'envoi de l'email: {e}")

def invite_user_to_project(username, project_name):
    """
        Invite un utilisateur à rejoindre un projet.

        Cette fonction récupère l'ID du projet et de l'utilisateur dans la base de données,
        puis envoie un email d'invitation à l'utilisateur avec les détails du projet.

    project = DB_API.get_project_id_by_project_name(project_name)
    userID = DB_API.get_user_id_by_username(username)
    email = DB_API.get_user_email(userID)
    permission = 1
    joined = 1

    if not project:
        print(f"Projet '{project_name}' introuvable.")
        return

        print(f"Utilisateur '{username}' introuvable.")
        return
    subject = f'Invitation à rejoindre le projet "{project_name}"'

        f'Bonjour,\n\nVous avez été invité à rejoindre le projet "{project_name}"'
        f".\n\n\nCordialement,\nL'équipe de gestion de projet."
    )

    send_email(subject, body, email)
    DB_API.add_user_to_group(project, userID, permission, joined)

def check_tasks():
    """
        Vérifie les tâches en fonction de leur date d'échéance et envoie des rappels par email.

        Cette fonction vérifie les tâches dans la base de données pour déterminer si leur date limite est passée ou approche.
        Si la tâche est en retard, un email est envoyé à l'utilisateur assigné. Si la date limite approche, un rappel est envoyé.
    """
    now = datetime.now()
    print(f'Vérification des tâches à {now.strftime("%d/%m/%Y %H:%M")}...')

    tasks = DB_API.get_all_tasks()
    colonnes = [
        "taskID",
        "title",
        "description",
        "status",
        "priority",
        "covered",
        "assign",
        "due_date",
    ]
    for task_tuple in tasks:
        task = dict(zip(colonnes, task_tuple))
        due_date = task["due_date"]
        time_remaining = (due_date - now).total_seconds()
        user_emails = DB_API.get_emails_by_task_project_id(task["assign"])
        if time_remaining < 0:
            for user_email in user_emails:
                subject = f'Tâche en retard : {task["title"]}'
                body = f"""Bonjour,

        La tâche "{task['title']}" est en retard. Veuillez la terminer dans les meilleurs délais.

        Cordialement,
        L'équipe de gestion de projet."""
                send_email(subject, body, user_email)
                print(f"Email envoyé à {user_email} concernant une tâche en retard.")

        elif 0 < time_remaining < 86400:  # Moins de 24 heures
            for user_email in user_emails:
                subject = f'Rappel : La date limite approche pour "{task["title"]}"'
                body = f"""Bonjour,

        N'oubliez pas que la tâche "{task['title']}"
        doit être terminée avant le {due_date.strftime('%d/%m/%Y %H:%M')}.

        Cordialement,
        L'équipe de gestion de projet."""
                send_email(subject, body, user_email)
                print(f"Rappel envoyé à {user_email}.")

def main():
    """
        Affiche un menu principal et permet d'exécuter diverses actions, telles que la gestion des utilisateurs, projets et tâches.

        Cette fonction propose à l'utilisateur un menu interactif pour créer des utilisateurs et des projets, inviter des utilisateurs,

    while True:

        print("2. Créer un projet")
        print("3. Inviter un utilisateur à un projet")
        print("4. Ajouter une tâche")
        print("5. Lister les projets")
        print("6. Lister les tâches d'un projet")
        print("7. Vérifier les tâches (envoyer des rappels)")
        print("8. Quitter")
        choice = input("Choisissez une option : ")

        if choice == "1":
            username = input("Nom d'utilisateur : ")
            email = input("Adresse e-mail : ")
            password = input("Mot de passe : ")
            gmail = "@gmail"
            if gmail in email:
                is_google_user = True
            else:
                is_google_user = False
            DB_API.add_user(username, password, email, is_google_user)
            print(f"Utilisateur {username} créé avec succès.")

        elif choice == "2":
            name = input("Nom du projet : ")
            DB_API.add_project(name)
            print(f"Projet '{name}' créé avec succès.")

        elif choice == "3":
            username = input("Nom de l'utilisateur : ")
            project_name = input("Nom du projet : ")
            invite_user_to_project(username, project_name)

        elif choice == "4":
            project_name = input("Nom du projet : ")
            task_title = input("Titre de la tâche : ")
            priority = input("Priorité (Haute, Moyenne, Basse) : ")
            if priority == "Haute":
                priority = 3
            elif priority == "Moyenne":
                priority = 2
            elif priority == "Basse":
                priority = 1
            else:
                print("erreur")
            description = input("Description : ")
            due_date_str = input("Date d'échéance (YYYY-MM-DD HH:MM) : ")
            username = input("Utilisateur assigné : ")
            userID = DB_API.get_user_id_by_username(username)
            task_projectID = DB_API.add_task_group()
            DB_API.add_user_to_task_group(task_projectID, userID)

def delete_user(username: str, user_id: Optional[int] = None) -> bool:
    """
    Permet de delete un utilisateur de la db et toutes ses dépendances.

    Cette fonction renvoie un boolean. En cas d'échec,
    elle renverra False et en cas de succes, elle renverra True.

    :param username: Username de l'utilisateur
    :param user_id: ID de l'utilisateur (optionnel)

    Exemple:
    >>> err = delete_user("toto")
    >>> if err:
    >>>     print("supprimé avec succes")
    >>> err = delete_user(user_id=4)
    >>> if err:
    >>>     print("supprimé avec succes")
    """
    if not user_id:
        user_id = DB_API.get_user_id_by_username(username)
        if isinstance(user_id, Exception):
            return False
    err = DB_API.delete_user(user_id)
    if isinstance(err, Exception):
        return False
    return True

def update_task_info(
    task_info_id: int,
    status: Optional[int] = None,  # noqa
    priority: Optional[int] = None,  # noqa
    title: Optional[str] = None,  # noqa
    description: Optional[str] = None,  # noqa
    dead_line: Optional[datetime] = None,  # noqa
    covered: Optional[bool] = None,  # noqa
    assign: Optional[int] = None,  # noqa
):
    """
    Met à jour une tâche dans la table tasks_infos.

    Cette fonction renvoie un boolean. En cas d'échec,
    elle renverra False et en cas de succes, elle renverra True.

    :param task_info_id: ID de la tâche à modifier.
    :param status: (optionnel) Nouveau statut de la tâche.
    :param priority: (optionnel) Nouvelle priorité de la tâche.
    :param title: (optionnel) Nouveau titre de la tâche.
    :param description: (optionnel) Nouvelle description de la tâche.
    :param dead_line: (optionnel) Nouvelle échéance (format datetime).
    :param covered: (optionnel) Nouvel état "covered" (booléen).
    :param assign: (optionnel) Nouvel assigné (ID de tasks_groups).

    Exemple:
    >>> success = update_task_info(task_info_id=4, title='nouveau title', priority=2, assign=2)
    >>> if success:
    >>>     print("supprimé avec succes")
    >>> success = update_task_info(task_info_id=4, status=2)
    >>> if success:
    >>>     print("supprimé avec succes")
    """
    _dict = {
        field: value
        for field, value in vars().items()
        if value is not None and field != "task_info_id"
    }

    err = DB_API.update_task_infos(task_info_id, _dict.keys(), _dict.values())
    if isinstance(err, Exception):
        return False
    return True
